from fastapi import APIRouter, HTTPException
from app.services.product_data_service import ProductDataService
from app.models.product import ProductResponse
from typing import List, Optional

router = APIRouter()
product_service = ProductDataService()

@router.get("/", response_model=List[ProductResponse])
async def get_products(
    limit: Optional[int] = 20,
    category: Optional[str] = None,
    search: Optional[str] = None
):
    """
    Retrieves a list of products based on optional filters.

    Args:
        limit: The maximum number of products to return (default: 20).
        category:  Filter products by a specific category (optional).
        search:  Search products based on a query string (optional).

    Returns:
        A list of ProductResponse objects.

    Raises:
        HTTPException:  If an error occurs during product retrieval.
    """
    try:
        products = await product_service.get_products(
            limit=limit,
            category=category,
            search=search
        )
        return products
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

@router.get("/categories")
async def get_categories():
    """Get available product categories"""
    try:

        return {"categories": categories}
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))
